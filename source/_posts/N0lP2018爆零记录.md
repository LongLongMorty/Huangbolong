	freopen("road.out","w",stdout);
	
	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d",&a[i]);
	}
	unsigned long long cnt=0;
	
	while(1){
		unsigned long long s=0;
		bool l0=0;
		bool flag=1;
		int i=1;
		while(i<=n){
			while(a[i]==0){
				++i;
				if(i>n)break;
			}
			if(i<=n){
				++s;
				while(a[i]!=0){
					--a[i];
					++i;
					if(i>n)break;
				}
			}
		}
		cnt+=s;
		if(s==0)break;
	}
	cout<<cnt<<'\n';

	fclose(stdout);

	for(int i=1;i<=n-1;++i){
		int u,v;
		scanf("%d%d",&u,&v);

		g[v].push_back(u);//建图

				if(*it==c2){
					ok=1;

				}
			}

				printf("-1\n");
				continue;
			}

			val+=b[c1].w;

			bx[c1]=1;//否则标记为不选

				if(b[g[c1][i]].w<ming){
					ming=b[g[c1][i]].w;
					mid=g[c1][i];
				}
			}
			val+=ming;
			s[mid]=1;
		}
		if(s2){//同s1

			s[c2]=1;
		}
